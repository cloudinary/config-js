# UploadPreset

## Example Usage

```typescript
import { UploadPreset } from "@cloudinary/config/models/components";

let value: UploadPreset = {
  headers: "X-Robots-Tag: noindex",
  moderation: "aws_rek_video",
  rawConvert: "google_speech:vtt:en-US",
  backgroundRemoval: "pixelz",
  format: "jpg",
  allowedFormats: "mp4,ogv,jpg,png,pdf",
  autoTagging: 0.5,
  detection: "coco_v2",
};
```
